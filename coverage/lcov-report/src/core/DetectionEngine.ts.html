function getRiskLevel(dangerScore: number): RiskLevel {
  if (dangerScore >= 90) return 'CRITICAL';
  if (dangerScore >= 71) return 'HIGH';
  if (dangerScore >= 31) return 'MEDIUM';
  return 'LOW';
}
